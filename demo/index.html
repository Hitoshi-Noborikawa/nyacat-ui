📝 使用例

<!-- 基本のNekoButton -->
<button class="btn-neko">クリックしてね</button>

<!-- 猫種別ボタン -->
<button class="btn-neko btn-neko-tuxedo">ハチワレ</button>
<button class="btn-neko btn-neko-orange">茶トラ</button>
<button class="btn-neko btn-neko-gray">グレー</button>
<button class="btn-neko btn-neko-calico">三毛猫</button>
<button class="btn-neko btn-neko-black">黒猫</button>
<button class="btn-neko btn-neko-white">白猫</button>

<!-- 猫パーツ付き -->
<button class="btn-neko btn-neko-tail">尻尾付き</button>
<button class="btn-neko btn-neko-paws">肉球付き</button>
<button class="btn-neko btn-neko-paws-css">CSS肉球</button>

<!-- NekoInput -->
<input type="text" class="input-neko" placeholder="基本の入力">
<input type="email" class="input-neko input-neko-error" placeholder="エラー状態">
<input type="text" class="input-neko input-neko-success" placeholder="成功状態">

<!-- 猫種バリエーション -->
<input type="text" class="input-neko input-neko-orange" placeholder="茶トラ風">
<input type="text" class="input-neko input-neko-tuxedo" placeholder="ハチワレ風">
<input type="text" class="input-neko input-neko-gray" placeholder="グレー風">

<!-- NekoLoading -->
<div class="loading-neko"></div>
<div class="loading-neko-groom"></div>

<!-- サイズバリエーション -->
<div class="loading-neko loading-neko-sm"></div>
<div class="loading-neko loading-neko-lg"></div>

<!-- ボタンとの組み合わせ -->
<button class="btn-neko flex items-center space-x-2" disabled>
  <div class="loading-neko-sm"></div>
  <span>送信中...</span>
</button>

<!-- カスタムカラー -->
<div class="bg-neko-pink text-neko-black p-4 rounded-paw">
  猫色の背景
</div>
